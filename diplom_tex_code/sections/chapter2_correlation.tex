\section{Корреляционный своп}
\label{sec:corr-swap}

\subsection{Определение, выплата, экономический смысл}
\label{sec:corr-swap-def}

Корреляционный своп (correlation swap)~--- внебиржевой контракт, по
которому в момент истечения сделки одна сторона получает разницу между
\emph{реализованной} средней парной корреляцией доходностей корзины
активов и заранее зафиксированным \emph{страйком}, умноженную на номинал.
Формально, для корзины из $n$ активов реализованная корреляция за период
$[0,T]$ задаётся как
\begin{equation}
  \rho_{\text{real}}
  \;=\; \frac{2}{n(n-1)}\sum_{1\le i<j\le n}\rho_{ij}^{\text{real}}(0,T),
\end{equation}
где $\rho_{ij}^{\text{real}}(0,T)$~--- парная реализованная корреляция
логдоходностей активов $i$ и $j$ на интервале $[0,T]$.

Контракт характеризуется набором параметров: номинал $N$, срок сделки
$T$, фиксированный страйк $K_{\text{fixed}}$. Выплата покупателю свопа
имеет вид
\begin{equation}
  \label{eq:corrswap-payoff}
  \text{Payoff}\;=\;N\bigl(\rho_{\text{real}} - K_{\text{fixed}}\bigr).
\end{equation}

  \centering

  \caption{Динамика активов, их корреляции и выплат корреляционного свопа}%
  \label{fig:corr_swap_dynamic}

Покупатель корреляционного свопа получает прибыль при росте средней
корреляции корзины и убыток при её падении. Это делает инструмент
естественным для торговли структурой кросс-зависимостей и для
хеджирования портфелей, чувствительных к режимным переходам
коррелированности.

\subsection{Расчёт справедливой стоимости}
\label{sec:corr-models}

В момент заключения сделки её справедливый страйк $K_{\text{fair}}$
определяется условием равенства нулю ожидания выплаты под риск-нейтральной
мерой $\Q$:
\begin{equation}
\label{eq:fair-strike}
  \E^{\Q}[\text{Payoff}] \;=\; N\bigl(\E^{\Q}[\rho_{\text{real}}]
                                   - K_{\text{fixed}}\bigr)\;=\;0
  \quad\Longrightarrow\quad
  K_{\text{fair}} \;=\; \E^{\Q}[\rho_{\text{real}}].
\end{equation}
В непрерывном пределе можно записать
\begin{equation}
\label{eq:rho-real-integral}
  \rho_{\text{real}} \;=\; \frac{1}{T}\int_0^T \rho_t\,\dd t,
  \qquad
  K_{\text{fair}} \;=\; \frac{1}{T}\int_0^T \E^{\Q}[\rho_t]\,\dd t.
\end{equation}
Дальнейший расчёт сводится к выбору модели динамики $\rho_t$ и
вычислению (или приближению) условного среднего $\E^{\Q}[\rho_t]$.

\paragraph{Модель с постоянной корреляцией.}
Простейшее предположение $\rho_t \equiv \rho$ даёт
\begin{equation}
\label{eq:fair-const}
  K_{\text{fair}} \;=\; \rho.
\end{equation}

\paragraph{Трёхпараметрическая модель возврата к среднему.}
Рассмотрим детерминированную динамику возврата к среднему
\begin{equation}
\label{eq:meanrev-ode}
  \dd \rho_t \;=\; \kappa(\bar\rho - \rho_t)\,\dd t.
\end{equation}
Решим~\eqref{eq:meanrev-ode} разделением переменных:
\begin{align*}
  \frac{\dd\rho_t}{\dd t} &= \kappa(\bar\rho - \rho_t),\\
  \frac{\dd\rho_t}{\bar\rho - \rho_t} &= \kappa\,\dd t,\\
  -\ln\!\bigl|\bar\rho - \rho_t\bigr| &= \kappa t + C,\\
  \bar\rho - \rho_t &= (\bar\rho - \rho_0)\,e^{-\kappa t},\\
  \rho_t &= \bar\rho + (\rho_0 - \bar\rho)\,e^{-\kappa t}.
\end{align*}
Подставим в определение $\rho_{\text{real}}$:
\begin{align*}
  \rho_{\text{real}}
  &= \frac{1}{T}\int_{0}^{T}\rho_t\,\dd t
  = \frac{1}{T}\int_{0}^{T}\Bigl[\bar\rho + (\rho_0-\bar\rho)e^{-\kappa t}\Bigr]\dd t \\
  &= \bar\rho + \frac{\rho_0-\bar\rho}{T}\int_{0}^{T}e^{-\kappa t}\,\dd t
  = \bar\rho + \frac{\rho_0-\bar\rho}{T}\cdot\frac{1-e^{-\kappa T}}{\kappa}.
\end{align*}
Поскольку траектория $\rho_t$ детерминирована, $K_{\text{fair}} = \E^{\Q}[\rho_{\text{real}}] = \rho_{\text{real}}$:
\begin{equation}
\label{eq:fair-meanrev}
  \boxed{\;K_{\text{fair}}^{\text{MR}}
  \;=\; \bar\rho + (\rho_0-\bar\rho)\,\frac{1-e^{-\kappa T}}{\kappa T}.\;}
\end{equation}

\paragraph{Стохастическая модель Якоби.}
Стохастическое дифференциальное уравнение
\begin{equation}
\label{eq:jacobi}
  \dd\rho_t \;=\; \kappa(\bar\rho-\rho_t)\,\dd t
                + \sigma\sqrt{1-\rho_t^{\,2}}\,\dd W_t
\end{equation}
обеспечивает $\rho_t \in (-1,1)$ при подходящих
параметрах~\cite{ahdida2013jacobi}. Возьмём математическое ожидание
обеих частей~\eqref{eq:jacobi} под мерой $\Q$; стохастический член
обращается в ноль, и
\[
  \E^{\Q}[\dd\rho_t] \;=\; \kappa(\bar\rho - \E^{\Q}[\rho_t])\,\dd t.
\]
Обозначая $m_t = \E^{\Q}[\rho_t]$, получаем линейное ОДУ
\[
  \frac{\dd m_t}{\dd t} \;=\; \kappa(\bar\rho - m_t),
\]
которое, как было показано выше при выводе формулы
\eqref{eq:fair-meanrev}, имеет решение
\[
  m_t \;=\; \bar\rho + (\rho_0 - \bar\rho)\,e^{-\kappa t}.
\]
Тогда
\begin{align*}
  K_{\text{fair}}^{\text{Jac}}
  &= \E^{\Q}[\rho_{\text{real}}]
   = \frac{1}{T}\int_{0}^{T} \E^{\Q}[\rho_t]\,\dd t
   = \frac{1}{T}\int_{0}^{T} m_t\,\dd t \\
  &= \frac{1}{T}\int_{0}^{T}\Bigl[\bar\rho + (\rho_0-\bar\rho)e^{-\kappa t}\Bigr]\dd t.
\end{align*}
Окончательно
\begin{equation}
\label{eq:fair-jacobi}
  \boxed{\;K_{\text{fair}}^{\text{Jac}}
  \;=\; \bar\rho + (\rho_0-\bar\rho)\,\frac{1-e^{-\kappa T}}{\kappa T}.\;}
\end{equation}
Замечание: для среднего стохастическая модель Якоби даёт тот же
$K_{\text{fair}}$, что и детерминированная модель возврата к среднему.
Различие проявляется уже на уровне второго момента (вариация
$\rho_{\text{real}}$) и при ценообразовании производных от $\rho$
выпуклых инструментов.

\paragraph{Стохастическая модель Фишера--OU.}
Преобразование Фишера $\rho_t = \tanh(x_t)$ гарантирует $\rho_t\in(-1,1)$
автоматически. Зададим $x_t$ как Орнштейн--Уленбековский процесс
\begin{equation}
\label{eq:fisher-ou}
  \dd x_t \;=\; \kappa(\bar x - x_t)\,\dd t + \eta\,\dd W_t.
\end{equation}
Стандартное решение СДУ~\eqref{eq:fisher-ou} (через интегрирующий
множитель $e^{\kappa t}$) даёт
\[
  x_t \;=\; \bar x + (x_0 - \bar x)\,e^{-\kappa t}
            + \eta\!\int_0^{t} e^{-\kappa(t-s)}\,\dd W_s.
\]
Поскольку стохастический интеграл --- центрированная гауссова
случайная величина:
\[
  m_t \;\equiv\; \E^{\Q}[x_t] \;=\; \bar x + (x_0 - \bar x)\,e^{-\kappa t},
\]
\[
  v_t \;\equiv\; \Var^{\Q}[x_t]
       \;=\; \eta^{2}\int_0^{t}\!e^{-2\kappa(t-s)}\,\dd s
       \;=\; \frac{\eta^{2}}{2\kappa}\bigl(1-e^{-2\kappa t}\bigr).
\]
Заметим, что
\(
  \E^{\Q}[\rho_t] \;=\; \E^{\Q}[\tanh(x_t)] \neq \tanh(\E^{\Q}[x_t])
\)
из-за выпуклости $\tanh$. Разложим $\tanh$ в окрестности $m_t$ во
втором порядке:
\[
  \tanh(x_t) \;\approx\; \tanh(m_t)
  + \sech^{2}(m_t)\,(x_t - m_t)
  - \sech^{2}(m_t)\,\tanh(m_t)\,(x_t - m_t)^{2}.
\]
Беря $\E^{\Q}$, используя $\E^{\Q}[x_t - m_t]=0$ и
$\E^{\Q}[(x_t - m_t)^{2}] = v_t$, получаем
\begin{equation}
\label{eq:fisher-Erho}
  \E^{\Q}[\rho_t]
  \;\approx\; \tanh(m_t)\bigl(1 - v_t\,\sech^{2}(m_t)\bigr).
\end{equation}
Окончательно,
\begin{equation}
\label{eq:fair-fisher}
  \boxed{\;K_{\text{fair}}^{\text{Fish}}
  \;\approx\; \frac{1}{T}\int_0^{T}
              \tanh(m_t)\,\bigl(1 - v_t\,\sech^{2}(m_t)\bigr)\,\dd t.\;}
\end{equation}
Интеграл~\eqref{eq:fair-fisher} в общем случае берётся численно (например,
методом Симпсона). При $\eta\to 0$ дисперсия $v_t \to 0$, и
формула~\eqref{eq:fair-fisher} переходит в детерминированную
$\frac{1}{T}\int_0^T \tanh(m_t)\dd t$.

\subsection{Forward-стоимость и аддитивность по уже реализованным
            значениям}
\label{sec:corr-fv}

Корреляционные свопы с дневной отрисовкой реализованной корреляции
обладают удобным свойством: справедливая стоимость свопа в произвольный
момент $t\in[0,T]$ распадается на детерминированный накопленный $P\&L$
по прошедшему отрезку $[0,t]$ и стоимость <<нового>> свопа на
остаточный срок $[t,T]$.

\begin{proposition}
\label{prop:corr-decomp}
Пусть на $[0,T]$ корреляционный своп с номиналом $N$ имеет страйк
$K_{\text{fixed}}$, а реализованная корреляция на интервале $[0,t]$ уже
известна и равна $\rho^{[0,t]}_{\text{real}}$. Тогда дисконтированная
forward-стоимость свопа в момент $t$ имеет вид
\begin{equation}
\label{eq:corr-decomp}
  V_t \;=\; e^{-r(T-t)}\biggl[
    \tfrac{t}{T}\bigl(\rho^{[0,t]}_{\text{real}}-K_{\text{fixed}}\bigr)
    + \tfrac{T-t}{T}\bigl(K_{\text{fair}}^{[t,T]}-K_{\text{fixed}}\bigr)
  \biggr]\,N,
\end{equation}
где $K_{\text{fair}}^{[t,T]} = \E_t^{\Q}\!\left[\tfrac{1}{T-t}\int_t^T\rho_s\,\dd s\right]$
--- справедливый страйк корреляционного свопа на остаточный срок.
\end{proposition}

\begin{proof}
По определению, стоимость свопа в момент $t$ --- это дисконтированное
условное ожидание выплаты:
\begin{equation}
\label{eq:Vt-def}
  V_t \;=\; e^{-r(T-t)}\,
           \E_t^{\Q}\!\bigl[\,N\bigl(\rho_{\text{real}}^{[0,T]}-K_{\text{fixed}}\bigr)\,\bigr].
\end{equation}
Разобьём интеграл, определяющий $\rho_{\text{real}}^{[0,T]}$, на участки
$[0,t]$ и $[t,T]$:
\begin{align}
  \rho_{\text{real}}^{[0,T]}
  &= \frac{1}{T}\int_{0}^{T}\rho_s\,\dd s
   = \frac{1}{T}\int_{0}^{t}\rho_s\,\dd s
   + \frac{1}{T}\int_{t}^{T}\rho_s\,\dd s \notag\\
  &= \frac{t}{T}\cdot\underbrace{\frac{1}{t}\int_{0}^{t}\rho_s\,\dd s}_{\rho_{\text{real}}^{[0,t]}}
   + \frac{T-t}{T}\cdot\underbrace{\frac{1}{T-t}\int_{t}^{T}\rho_s\,\dd s}_{\rho_{\text{real}}^{[t,T]}}.
\label{eq:rho-split}
\end{align}
Первое слагаемое в~\eqref{eq:rho-split} в момент $t$ известно
\emph{детерминированно} (это уже произошедшая часть истории), второе
--- случайно. Применяем линейность условного ожидания:
\begin{align*}
  \E_t^{\Q}\!\bigl[\rho_{\text{real}}^{[0,T]}\bigr]
  &= \frac{t}{T}\,\rho_{\text{real}}^{[0,t]}
   + \frac{T-t}{T}\,\E_t^{\Q}\!\bigl[\rho_{\text{real}}^{[t,T]}\bigr] \\
  &= \frac{t}{T}\,\rho_{\text{real}}^{[0,t]}
   + \frac{T-t}{T}\,K_{\text{fair}}^{[t,T]},
\end{align*}
где использовано определение справедливого страйка свопа на остаточный
срок: $K_{\text{fair}}^{[t,T]} = \E_t^{\Q}[\rho_{\text{real}}^{[t,T]}]$.
Подставляя в~\eqref{eq:Vt-def}, получаем
\begin{align*}
  V_t
  &= e^{-r(T-t)}\,N\!\left(
       \frac{t}{T}\,\rho_{\text{real}}^{[0,t]}
     + \frac{T-t}{T}\,K_{\text{fair}}^{[t,T]}
     - K_{\text{fixed}}
     \right).
\end{align*}
Запишем $K_{\text{fixed}} = \tfrac{t}{T}K_{\text{fixed}} + \tfrac{T-t}{T}K_{\text{fixed}}$
и сгруппируем слагаемые с одинаковыми весами:
\begin{align*}
  V_t
  &= e^{-r(T-t)}\,N\!\left(
       \tfrac{t}{T}\bigl(\rho_{\text{real}}^{[0,t]} - K_{\text{fixed}}\bigr)
     + \tfrac{T-t}{T}\bigl(K_{\text{fair}}^{[t,T]} - K_{\text{fixed}}\bigr)
     \right).
\end{align*}
Это в точности~\eqref{eq:corr-decomp}.
\end{proof}

\subsection{Хеджирование от пузырей с помощью корреляционных свопов}
\label{sec:corr-hedge-bubble}

\paragraph{Эмпирическая мотивация.}
Кризисные эпизоды на финансовых рынках систематически сопровождаются
\emph{ростом} корреляций между секторами и активами~--- этот эффект
получил название <<correlation breakdown>>. Кризис 2000~года (dot-com
bubble) и кризис 2008~года~--- классические примеры: падение широкого
рынка сопровождается заметным ростом средней парной корреляции внутри
крупных индексов. На этом наблюдении основана естественная стратегия
хеджирования: \emph{покупка} корреляционных свопов эквивалентна покупке
защиты от системных эпизодов одновременного снижения активов.

\paragraph{Постановка задачи и роли параметров.}
В текущей задаче можно явно выделив три уровня параметров:
\begin{itemize}
  \item \textbf{Входные данные клиента --- портфель базовых активов
        $w$.}
        В каждый момент времени клиент держит фиксированный портфель
        весов $w = (w_1,\dots,w_n)$ на корзине из $n$ активов. Состав
        $w$ выбирается клиентом независимо от хеджирования: обычно это
        результат портфельной оптимизации (теории
        Марковица)~\cite{markowitz1952portfolio} с учётом ожидаемых
        доходностей и ковариационной матрицы при <<нормальном>>
        режиме рынка. Для хеджера $w$ --- внешний вход.
  \item \textbf{Гиперпараметры --- срочности корреляционных свопов
        $\{T_k\}_{k=1}^{K}$.}
        Набор срочностей фиксируется заранее. На практике он диктуется
        \emph{ликвидностью}: на внебиржевом рынке корреляционных
        свопов доступны лишь стандартные срочности (как правило 3, 6
        и 12 месяцев), и торговать <<своими>> срочностями
        нерационально.
  \item \textbf{Модельные переменные --- номиналы свопов
        $\{N_k\}_{k=1}^{K}$.}
        Это единственный набор параметров, который остаётся
        свободным и подбирается решением оптимизационной задачи.
\end{itemize}
Хеджевый портфель на горизонте $T_{\max} = \max_k T_k$ строится как
набор корреляционных свопов с номиналами $\{N_k\}$. Обозначим
$\Delta V_{\text{port}}^{\text{stress}}$ убыток клиентского портфеля в
стрессовом сценарии и $\Delta V_k^{\text{stress}}$ изменение стоимости
$k$-го свопа в том же сценарии. Номиналы $N_k$ подбираются
минимизацией оста\-точной вариации $P\&L$:
\begin{equation}
\label{eq:hedge-opt}
  \min_{N_1,\dots,N_K} \;\;
  \E\!\left[\Bigl(\Delta V_{\text{port}}^{\text{stress}}(w)
                  + \textstyle\sum_{k=1}^{K} N_k\,\Delta V_k^{\text{stress}}\Bigr)^{\!2}\right],
\end{equation}
где математическое ожидание берётся по эмпирическому распределению
стрессовых окон (исторические кризисные периоды). Поскольку $w$
зафиксирован клиентом, а $\{T_k\}$ --- ликвидностью, единственные
свободные переменные --- $\{N_k\}$, и задача~\eqref{eq:hedge-opt}
является \emph{выпуклой} квадратичной оптимизацией без ограничений
(или с линейными ограничениями вида $N_k \ge 0$, если допускается
только лонг-позиция в корреляции).

на практике очень сложно провести численную оптимизацию портфеля, поэтому можно

оптимальные параметры $N_k$ на основе этих данных оптимизируя коэффициенты доходнотси:
Шарпа~\cite{Sharpe1966} и Мартина~\cite{MartinMcCann1989}, т.к. именно эти коээфициенты 
помогут не только учесть хеджирующую состовляющую портфеля, но и рост волатильности портфеля 
за счет переоценки деривативов. 

\subsection{Корреляционный своп как источник волатильности портфеля}
\label{sec:corr-source-vol}

Корреляционный своп как защита от пузырей работает не <<бесплатно>>.
В спокойные периоды реализованная корреляция корзины может оказаться
ниже зафиксированного страйка $K_{\text{fixed}}$, и покупатель свопа
несёт регулярный отрицательный $P\&L$ ---  премию за страховку.
Кроме того, ошибка модели корреляции $\rho_t$ (например, неверная
оценка параметров $\kappa,\bar\rho$) или режимный сдвиг структуры
зависимости приводят к дополнительной волатильности $P\&L$,
которая в нехеджевом сценарии отсутствовала бы.

\subsection{Хеджирование rainbow-опционов с помощью корреляционных
           свопов}
\label{sec:corr-hedge-rainbow}

Идея, лежащая в основе хеджирования rainbow-опциона с помощью
корреляционных свопов, проста: \emph{цена rainbow-опциона зависит от
парных корреляций базовых активов}, а корреляционный своп есть
инструмент, чьё значение \emph{линейно} реагирует на изменение
реализованной корреляции. Если выделить из выплаты rainbow-опциона
компоненту, чувствительную к корреляции, и реплицировать её
динамическим набором корр-свопов, исходный риск можно перенести на
ликвидный инструмент.

Формально, цена rainbow-опциона $C_{\text{rb}}$ является функцией
многих риск-факторов: цен активов $S^{(i)}$, их волатильностей
$\sigma_i$ и парных корреляций $\rho_{ij}$. Разложение в окрестности
средней корреляции $\bar\rho$ имеет вид
\begin{equation}
\label{eq:rb-rho-expand}
  C_{\text{rb}}(\rho) \;\approx\;
  C_{\text{rb}}(\bar\rho)
   + \underbrace{\frac{\partial C_{\text{rb}}}{\partial \rho}
                 \bigg|_{\bar\rho}}_{\text{корреляционная вега}}\,
                 (\rho - \bar\rho)
   + O\bigl((\rho-\bar\rho)^{2}\bigr).
\end{equation}
Линейная часть~\eqref{eq:rb-rho-expand} точно реплицируется одним
корреляционным свопом с номиналом
$N = \tfrac{\partial C_{\text{rb}}}{\partial \rho}$ и страйком
$K_{\text{fixed}} = \bar\rho$: его выплата в момент $T$ равна
$N(\rho_{\text{real}} - \bar\rho)$, что совпадает с линейным
приращением цены rainbow-опциона по корреляции. Нелинейная часть
$O\bigl((\rho-\bar\rho)^{2}\bigr)$ остаётся остаточным риском,
которому посвящена \cref{sec:rainbow-decomp}.

Динамическая стратегия хеджирования rainbow-опциона состоит из трёх
блоков. Во-первых, дельта-хедж по базовым активам устраняет
чувствительность к их ценам $S^{(i)}$. Во-вторых, вега-хедж ванильными
опционами устраняет чувствительность к волатильностям $\sigma_i$.
В-третьих, корреляционный своп устраняет линейную часть
чувствительности к парным корреляциям $\rho_{ij}$. Подробности
декомпозиции цены и оста\-точная нелинейная часть рассматриваются в
\cref{sec:rainbow-decomp}.

\section{Rainbow-опционы}
\label{sec:rainbow-section}

\subsection{Полу-замкнутая формула и сложность её применения}
\label{sec:rainbow-formula}

Для двух активов в модели Блэка--Шоулза существует полу-аналитическое
выражение для цены rainbow-опционов вида $\max(S^{(1)},S^{(2)})$ и
$\min(S^{(1)},S^{(2)})$ через двухмерную функцию нормального
распределения~\cite{stulz1982options}:
\[
  \text{Обозначим за }C_{\max}(S^{(1)},S^{(2)};K,T) \text{--- цену Call on Max опциона}
\]
\begin{equation}
\label{eq:rainbow-formula}
  C_{\max}(S^{(1)},S^{(2)};K,T)
  \;=\; S^{(1)}\Phi_2(\cdot) + S^{(2)}\Phi_2(\cdot) - K e^{-rT}\Phi_2(\cdot),
\end{equation}
Где \[
\Phi_2(a,b;\rho)
=
\mathbb{P}(X \le a,\; Y \le b)
\]
где случайный вектор \((X,Y)\) имеет двумерное нормальное распределение:
\[
X,Y \sim \mathcal{N}(0,1),
\qquad
\mathrm{Corr}(X,Y)=\rho.
\]
Функция \(\Phi_2(a,b;\rho)\) называется двумерной функцией распределения стандартного нормального закона и задаётся интегралом
\[
\Phi_2(a,b;\rho)
=
\int_{-\infty}^{a}
\int_{-\infty}^{b}
\frac{1}{2\pi\sqrt{1-\rho^2}}
\exp\left(
-\frac{x^2-2\rho xy+y^2}{2(1-\rho^2)}
\right)
\,dy\,dx.
\] ~\cite{plackett1954}

\begin{figure}[H]
  \centering
  \includegraphics[width=0.95\linewidth]{pictures/Phi_distribution.png}
  \caption{Графики плотности двумерного нормального распределения для случайных
   величин с разной корреляцией $\rho$.}%
  \label{fig:phi_distribution}
\end{figure}

Точная формула оценки цены радужного опциона Call on Max для двух
активов~\cite{stulz1982options, johnson1987options}: для срока до
погашения $\tau = T-t$ цена такого опциона в момент $t$ выражается
формулой
\begin{equation}
\label{eq:rainbow-stulz}
\begin{aligned}
  C_{\max}(t)
  \;=\;& S_1(t)\,e^{-q_1\tau}\,\Phi_2\!\bigl(y_1,\;d;\;\rho_1\bigr) \\
   +\;& S_2(t)\,e^{-q_2\tau}\,\Phi_2\!\bigl(y_2,\;-d+\sigma\sqrt{\tau};\;\rho_2\bigr) \\
   -\;& K\,e^{-r\tau}\,
        \Bigl[\,1 - \Phi_2\!\bigl(-y_1+\sigma_1\sqrt{\tau},\;
                              -y_2+\sigma_2\sqrt{\tau};\;\rho\bigr)\,\Bigr],
\end{aligned}
\end{equation}
где
\begin{itemize}
  \item $S_1(t),\,S_2(t)$~--- текущие цены базовых активов;
  \item $K$~--- страйк, $T$~--- срок исполнения, $\tau = T-t$~--- время
        до погашения;
  \item $r$~--- безрисковая процентная ставка;
  \item $q_1,\,q_2$~--- непрерывные дивидендные доходности (или
        стоимости поддержания позиции, cost-of-carry) по активам~1
        и~2; для бездивидендных активов $q_i = 0$;
  \item $\sigma_1,\,\sigma_2$~--- мгновенные волатильности логарифмов
        цен $\ln S_1$ и $\ln S_2$ соответственно;
  \item $\rho$~--- мгновенная парная корреляция между приращениями
        $\dd \ln S_1$ и $\dd \ln S_2$.
\end{itemize}

\paragraph{Вспомогательная волатильность $\sigma$.}
Введём волатильность \emph{отношения} активов:
\begin{equation}
\label{eq:rainbow-sigma}
  \sigma \;=\; \sqrt{\sigma_1^{\,2} - 2\rho\,\sigma_1\sigma_2 + \sigma_2^{\,2}}.
\end{equation}
Это волатильность процесса $\ln(S_1/S_2)$ под мерой $\Q$; именно она
управляет распределением спреда $S_1-S_2$ и появляется в формуле
Маргрэйба~\cite{margrabe1978value}.

\paragraph{Стандартизованные лог-моменты $y_1,\,y_2$.}
\begin{equation}
\label{eq:rainbow-y}
  y_1
  \;=\;\frac{\ln(S_1/K) + \bigl(r - q_1 + \tfrac{1}{2}\sigma_1^{\,2}\bigr)\tau}
            {\sigma_1\sqrt{\tau}},
  \qquad
  y_2
  \;=\;\frac{\ln(S_2/K) + \bigl(r - q_2 + \tfrac{1}{2}\sigma_2^{\,2}\bigr)\tau}
            {\sigma_2\sqrt{\tau}}.
\end{equation}
По форме $y_i$~--- это в точности аргумент $d_1$ формулы Блэка--Шоулза
для $i$-го актива со страйком $K$. Аналогично, $-y_i+\sigma_i\sqrt{\tau}$~---
аргумент $-d_2$ Блэка--Шоулза.

\paragraph{Стандартизованный лог-спред $d$.}
\begin{equation}
\label{eq:rainbow-d}
  d \;=\; \frac{\ln(S_1/S_2) + \bigl(q_2 - q_1 + \tfrac{1}{2}\sigma^{2}\bigr)\tau}
              {\sigma\sqrt{\tau}}.
\end{equation}
Аргумент $d$ выражает, насколько <<впереди>> идёт актив~1 по сравнению
с активом~2; он возникает благодаря тождеству
$\max(S_1,S_2) = S_1 + (S_2-S_1)^{+}$, на котором основан вывод
формулы~\eqref{eq:rainbow-stulz}.

\paragraph{Промежуточные корреляции $\rho_1, \rho_2$.}
Двумерная нормальная функция $\Phi_2$ в каждом из трёх слагаемых
\eqref{eq:rainbow-stulz} использует свою корреляцию. В первом
слагаемом аргументы~--- стандартизованные $y_1$ (по логарифму
актива~1) и $d$ (по лог-спреду $\ln(S_1/S_2)$), и соответствующая
корреляция между ними равна
\begin{equation}
\label{eq:rainbow-rho1}
  \rho_1 \;=\; \frac{\sigma_1 - \rho\,\sigma_2}{\sigma}.
\end{equation}
Аналогично, во втором слагаемом~--- $y_2$ и $-d+\sigma\sqrt{\tau}$, и
\begin{equation}
\label{eq:rainbow-rho2}
  \rho_2 \;=\; \frac{\sigma_2 - \rho\,\sigma_1}{\sigma}.
\end{equation}
Третье слагаемое~--- это $\Q$-вероятность того, что оба актива в момент
$T$ окажутся ниже страйка $K$ (и опцион ничего не платит); в нём
используется исходная корреляция $\rho$.

Для $n\ge 3$ активов аналитические выражения
становятся неудобны, а необходимость пересчёта при изменении $\rho$
делает прямую формулу плохо приспособленной для практического
динамического хеджирования. Поэтому на практике используется
\emph{декомпозиция} цены.

\subsection{Декомпозиция цены на компоненты}
\label{sec:rainbow-decomp}

Для начала поговорим об опционе best-of, с формулой выплаты:
\[
  \text{Payoff}_{best-of} = \max(S^{(1)}_T, S^{(2)}_T, K). 
\]
Но payoff такого опциона, так же как и его цену можно легко свести к опциону Call on Max:
\begin{equation}
\label{eq:best-of_to_call_to_max}
\begin{aligned}
\text{Payoff}_{best-of}
&= \max(S^{(1)}_T, S^{(2)}_T, K) \\
&= \max(\max(S^{(1)}_T, S^{(2)}_T), K) \\
&= K + \max(\max(S^{(1)}_T, S^{(2)}_T) - K, 0) \\
&= K + \text{Payoff}_{max},
\end{aligned}
\end{equation}

\begin{equation}
C_{best-of}(S^{(1)}, S^{(2)}; K, T)
=
K e^{-rT}
+
C_{\max}(S^{(1)}, S^{(2)}; K, T).
\end{equation}
Далее будем оценивать и декомпозировать именно $C_{\max}$.
Воспользуемся тождеством
\begin{equation}
\label{eq:max_to_plus_transform}
\begin{aligned}
(\max(S_1,S_2) - K)^{+}
&= (S_1 - K)1_{S_1 > S_2, S_1 > K} + (S_2 - K)1_{S_2 > S_1, S_2 > K} \\
&= S_1 \cdot 1_{S_1 > S_2, S_1 > K} + \; S_2 \cdot 1_{S_2 > S_1, S_2 > K} - K\cdot 1_{\max(S_1,S_2) > K} \\
\end{aligned}
\end{equation}

Далее можно воспользоватсья переход далее мы делаем переход мер и получаем формулу Штульца,
которую мы описывали выше - это ключевой шаг выделения линейных компонент. \\
Далее через равенство греков мы найдем части линейные по волатильности и по корреляции:

\begin{equation}
% \label{eq:C_max_linear_decomposition}
\begin{aligned}
C_{max}(t) 
&= a_0(t)+ a_1(t) S_1(t) + a_2(t) S_2(t) + b(t) CorrSwap(t) \\
&+ \sum_i c_i(t) \cdot \text{Call}_i(t) + \sum_i d_i(t) \cdot \text{Put}_i(t) \\
a_1(t) &= \frac{\partial C_{\max}}{\partial S_1} = exp(-q_1 tau) \Phi_2(y_1, d, rho_1) \\
a_2(t) &= \frac{\partial C_{\max}}{\partial S_2} = exp(-q_2 tau) \Phi_2(y_2, d, rho_2) \\
\end{aligned}
\end{equation}
Далее расчитываем vega у радужного опциона и vega у ваниллы:

\begin{equation}
\begin{aligned}
c_1 &= (\frac{\partial C_{\max}}{\partial \sigma_1}) / \nu_1 \text{, где } \nu_1 = \frac{\partial C_1^{BS}}{\partial \sigma_1} \\
c_2 &= (\frac{\partial C_{\max}}{\partial \sigma_2}) / \nu_2 \text{, где } \nu_2 = \frac{\partial C_2^{BS}}{\partial \sigma_2} \\
\end{aligned}
\end{equation}
Фактически сколько ванилла call на $S_i$ нужно купить чтобы воспроизветси exposure радужного 
опциона по $\sigma_i$, или что тоже самое что купить vega через торгуемые ваниллы, так как 
vega самого радужного опциона не торгуется. \\
Далее расчитываем cega радужного опциона и cega корреляционного свопа,
 и находим сколько корреляционных свопов нужно купить, чтобы воспроизвести корреляционную 
 cega радужного опциона:

\begin{equation}
\begin{aligned}
cega &= (\frac{\partial C_{\max}}{\partial \rho})  \\
     &= \frac{\partial S_1(t)\,e^{-q_1\tau}\,\Phi_2\!\bigl(y_1,\;d;\;\rho_1\bigr) }{\partial \rho} \\
     &+ \frac{\partial S_2(t)\,e^{-q_2\tau}\,\Phi_2\!\bigl(y_2,\;-d + \sigma \sqrt{\tau};\;\rho_2\bigr) }{\partial \rho} \\
     &+ \frac{\partial K\,e^{-r\tau}\,
        \Bigl[\,1 - \Phi_2\!\bigl(-y_1+\sigma_1\sqrt{\tau},\;
                              -y_2+\sigma_2\sqrt{\tau};\;\rho\bigr)\,\Bigr] }{\partial \rho}\\
\frac{\partial S_1(t)\,e^{-q_1\tau}\,\Phi_2\!\bigl(y_1,\;d;\;\rho_1\bigr) }{\partial \rho} \\
&= S_1 exp(-q_1 \tau) \frac{\partial \Phi_2(y_1, d, \rho_1)}{\partial \rho} \\
&= S_1 exp(-q_1 \tau) \cdot ( (\Phi_2)_{y_1}(y1, d, \rho_1) \frac{\partial y_1}{\partial \rho} \\
&+ (\Phi_2)_d(y_1, d, \rho_1) \frac{\partial d}{\partial \rho} \\ 
&+ (\Phi_2)_{rho_1}(y_1, d, \rho_1) \frac{\partial \rho_1}{\partial \rho}) \\ 
&= S_1 exp(-q_1 \tau) \cdot (0 + \Phi_1(d) * N(\frac{y_1 - \rho_1 d}{\sqrt{1 - \rho_1^2}}) \frac{\partial d}{\partial \rho} \\
&+ \Phi_2(y_1, d, \rho_1) \frac{\partial \rho_1}{\partial \rho}) \\              
\end{aligned}
\end{equation}

\subsection{Динамическое хеджирование}
\label{sec:rainbow-dynhedge}

Динамическое хеджирование rainbow-опциона состоит в построении портфеля
$\Pi_t$ из базовых активов, ванильных опционов и корреляционных свопов,
коэффициенты которых пересчитываются в каждый момент $t$ так, чтобы
производные $\Pi_t$ по всем риск-факторам совпадали с производными
$C_{\text{rb}}$:
\begin{equation}
\label{eq:dyn-hedge}
  \frac{\partial \Pi_t}{\partial S^{(i)}_t} = \frac{\partial C_{\text{rb}}}{\partial S^{(i)}_t},
  \quad
  \frac{\partial \Pi_t}{\partial \sigma_i} = \frac{\partial C_{\text{rb}}}{\partial \sigma_i},
  \quad
  \frac{\partial \Pi_t}{\partial \rho_{ij}} = \frac{\partial C_{\text{rb}}}{\partial \rho_{ij}}.
\end{equation}
Ключевая проблема~--- транзакционные издержки на ребалансировку:
бесконечно частая корректировка экономически невозможна, поэтому на
практике используют дискретную сетку моментов ребалансировки и
учитывают спред/комиссии (см.~\cref{ch:results}).
